FUNZIONI UTILI DI STRING.H
strlen --> Calcola la lunghezza di una stringa (escluso il carattere null \0) --> size_t strlen(const char *str);
strcpy --> Copia una stringa in un'altra (incluso il carattere null \0) --> char *strcpy(char *dest, const char *src);
strncpy --> Copia al massimo n caratteri di una stringa in un'altra --> char *strncpy(char *dest, const char *src, size_t n);
strcat --> Concatena due stringhe. Aggiunge src alla fine di dest --> char *strcat(char *dest, const char *src);
strncat --> Concatena al massimo n caratteri di una stringa alla fine di un'altra --> char *strncat(char *dest, const char *src, size_t n);
strcmp --> Confronta due stringhe. Restituisce un valore negativo, zero o positivo --> int strcmp(const char *str1, const char *str2);
strchr --> Cerca la prima occorrenza di un carattere in una stringa --> char *strchr(const char *str, int c);
strrchr --> Cerca l'ultima occorrenza di un carattere in una stringa --> char *strrchr(const char *str, int c);
strstr --> Cerca la prima occorrenza di una sottostringa all'interno di una stringa --> char *strstr(const char *haystack, const char *needle);
strtok --> Divide una stringa in sottostringhe (token) utilizzando delimitatori --> char *strtok(char *str, const char *delim);
